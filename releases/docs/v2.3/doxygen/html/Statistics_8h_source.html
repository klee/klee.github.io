76 inline void StatisticManager::incrementStatistic(Statistic &s,
77 uint64_t addend) {
78 if (enabled) {
79 globalStats[s.id] += addend;
80 if (indexedStats) {
81 indexedStats[index*stats.size() + s.id] += addend;
